package org.example.room;

import org.example.Guest;

import java.util.Objects;

/**
 * Класс, представляющий одноместный номер в гостинице.
 * Этот класс предназначен для номеров, в которых может разместиться только один гость.
 */
public class SingleBedRoom extends Room {

    private static final int PRICE = 10000;
    private static final int ROOM_SIZE = 1;

    private Guest guest = null;

    /**
     * Конструктор для создания одноместного номера.
     * Принимает идентификатор номера.
     *
     * @param roomId Идентификатор номера.
     */
    public SingleBedRoom(int roomId) {
        super(roomId);
    }

    /**
     * Возвращает тип номера.
     *
     * @return Тип номера (SINGLE).
     */
    @Override
    public RoomType getRoomType() {
        return RoomType.SINGLE;
    }

    /**
     * Возвращает цену за номер.
     *
     * @return Суточная цена за номер (10000).
     */
    @Override
    public int getPrice() {
        return PRICE;
    }

    /**
     * Возвращает размер комнаты (в данном случае всегда 1).
     *
     * @return Размер комнаты (1).
     */
    @Override
    public int getRoomSize() {
        return ROOM_SIZE;
    }

    /**
     * Возвращает количество гостей, зарегистрированных в номере.
     * Для одноместного номера количество гостей может быть только 0 или 1.
     *
     * @return Количество гостей в номере (0 или 1).
     */
    @Override
    public int getNumberOfGuests() {
        return guest == null ? 0 : 1;
    }

    /**
     * Регистрирует гостя в одноместном номере.
     * Если номер уже занят, регистрация не происходит.
     *
     * @param guest Гость, которого нужно зарегистрировать.
     * @throws RuntimeException если номер уже занят.
     */
    @Override
    public void checkInGuest(Guest guest) {
        if (!this.isFree()) {
            throw new RuntimeException("Не удалось зарегистрировать гостя в номере " + this.getRoomId() + ". Номер занят!");
        }
        guest.setRoomId(this.roomId);
        this.guest = guest;
        this.isFree = false;
    }

    /**
     * Выселяет гостя из одноместного номера.
     * Если гость не зарегистрирован в номере, выбрасывается исключение.
     *
     * @param guest Гость, которого необходимо выселить.
     * @throws RuntimeException если гость не зарегистрирован в номере.
     */
    @Override
    public void evictGuest(Guest guest) {
        if (!this.guest.equals(guest)) {
            throw new RuntimeException("Не удалось выселить гостя " + guest.getGuestId() + " из номера " +
                    this.getRoomId() + ". Гость в этом номере не живет!");
        }
        this.guest = null;
        this.isFree = true;
    }

    /**
     * Возвращает строковое представление информации об одноместном номере.
     * Строка содержит ID номера, его статус (свободен/занят), тип, цену и информацию о госте.
     *
     * @return Строковое представление информации о номере.
     */
    @Override
    public String toString() {
        return "Номер комнаты : " + roomId +
                "\nстатус : " + (isFree ? "свободна" : "занята") +
                "\nтип : " + getRoomType().getDescription() +
                "\nсуточный тариф : " + getPrice() +
                "\nгость : " + (Objects.isNull(guest) ? "отсутствует" : guest.toString());
    }
}
