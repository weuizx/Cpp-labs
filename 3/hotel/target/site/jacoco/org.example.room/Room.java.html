package org.example.room;

import org.example.Guest;

/**
 * Абстрактный класс, представляющий номер в гостинице.
 * Содержит общую информацию о номере, такую как его ID и статус (свободен/занят),
 * а также предоставляет методы для работы с типом номера, ценой, размером и количеством гостей.
 */
public abstract class Room {

    protected final int roomId;
    protected boolean isFree;

    /**
     * Конструктор для создания нового номера.
     *
     * @param roomId Идентификатор номера.
     */
    public Room(int roomId) {
        this.roomId = roomId;
        this.isFree = true;
    }

    /**
     * Получить идентификатор номера.
     *
     * @return Идентификатор номера.
     */
    public int getRoomId() {
        return roomId;
    }

    /**
     * Проверить, свободен ли номер.
     *
     * @return true, если номер свободен, иначе false.
     */
    public boolean isFree() {
        return isFree;
    }

    /**
     * Получить тип номера.
     *
     * @return Тип номера.
     */
    public abstract RoomType getRoomType();

    /**
     * Получить цену номера.
     *
     * @return Цена номера.
     */
    public abstract int getPrice();

    /**
     * Получить размер номера.
     *
     * @return Размер номера в квадратных метрах.
     */
    public abstract int getRoomSize();

    /**
     * Получить количество гостей, которых может принять номер.
     *
     * @return Количество гостей.
     */
    public abstract int getNumberOfGuests();

    /**
     * Зарегистрировать гостя в номер.
     *
     * @param guest Гость, которого нужно зарегистрировать.
     */
    public abstract void checkInGuest(Guest guest);

    /**
     * Выселить гостя из номера.
     *
     * @param guest Гость, которого нужно выселить.
     */
    public abstract void evictGuest(Guest guest);

    /**
     * Получить строковое представление номера.
     *
     * @return Строка, представляющая номер.
     */
    public abstract String toString();
}
