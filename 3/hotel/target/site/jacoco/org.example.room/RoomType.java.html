package org.example.room;

/**
 * Перечисление типов номеров в гостинице.
 * Содержит три типа номеров:
 * 1. ЛЮКС (LUXURY) - номер люкс.
 * 2. ОДНОМЕСТНЫЙ (SINGLE) - одноместный номер.
 * 3. МНОГОМЕСТНЫЙ (MULTI) - многоместный номер.
 */
public enum RoomType {

    /**
     * Номер люкс.
     */
    LUXURY("Номер люкс"),

    /**
     * Одноместный номер.
     */
    SINGLE("Одноместный номер"),

    /**
     * Многоместный номер.
     */
    MULTI("Многоместный номер");

    private final String description;

    /**
     * Конструктор для создания типа номера с описанием.
     *
     * @param description Описание типа номера.
     */
    RoomType(String description) {
        this.description = description;
    }

    /**
     * Получает описание типа номера.
     *
     * @return Описание типа номера.
     */
    public String getDescription() {
        return description;
    }
}
