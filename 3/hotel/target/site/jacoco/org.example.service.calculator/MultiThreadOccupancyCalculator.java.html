package org.example.service.calculator;

import org.example.room.Room;

import java.util.Iterator;
import java.util.Map;
import java.util.concurrent.Callable;
import java.util.concurrent.ExecutorService;
import java.util.concurrent.Executors;
import java.util.concurrent.Future;

/**
 * Класс для расчета занятости номеров с использованием многопоточности.
 * Этот класс реализует интерфейс {@link OccupancyCalculator} и использует три потока для параллельной обработки
 * разных типов номеров (одноместные, многоместные, люксовые).
 */
public class MultiThreadOccupancyCalculator implements OccupancyCalculator {

    private final Map<Integer, Room> SINGLE_BED_ROOMS; // Карта с одноместными номерами
    private final Map<Integer, Room> MULTI_BED_ROOMS;  // Карта с многоместными номерами
    private final Map<Integer, Room> LUXURY_ROOMS;    // Карта с люксовыми номерами

    /**
     * Конструктор для создания экземпляра калькулятора с многопоточной обработкой занятости.
     *
     * @param SINGLE_BED_ROOMS Карта с одноместными номерами.
     * @param MULTI_BED_ROOMS Карта с многоместными номерами.
     * @param LUXURY_ROOMS Карта с люксовыми номерами.
     */
    public MultiThreadOccupancyCalculator(Map<Integer, Room> SINGLE_BED_ROOMS,
                                          Map<Integer, Room> MULTI_BED_ROOMS,
                                          Map<Integer, Room> LUXURY_ROOMS) {
        this.SINGLE_BED_ROOMS = SINGLE_BED_ROOMS;
        this.MULTI_BED_ROOMS = MULTI_BED_ROOMS;
        this.LUXURY_ROOMS = LUXURY_ROOMS;
    }

    /**
     * Метод для расчета текущей занятости номеров. Занятость вычисляется как отношение
     * количества занятых мест к общему количеству мест в номерах.
     * Многопоточное выполнение для разных типов номеров.
     *
     * @return Занятость в виде числа с плавающей точкой от 0.0 до 1.0.
     *         Например, если занято 50% мест, метод вернет 0.5.
     */
    @Override
    public float getOccupancy() {
        ExecutorService executor = Executors.newFixedThreadPool(3);

        Future<int[]> singleBedOccupancy = executor.submit(new RoomOccupancyTask(SINGLE_BED_ROOMS));
        Future<int[]> multiBedOccupancy = executor.submit(new RoomOccupancyTask(MULTI_BED_ROOMS));
        Future<int[]> luxuryOccupancy = executor.submit(new RoomOccupancyTask(LUXURY_ROOMS));

        int numberOfGuests;
        int numberOfBeds;
        try {
            numberOfGuests = singleBedOccupancy.get()[0] + multiBedOccupancy.get()[0] + luxuryOccupancy.get()[0];
            numberOfBeds = singleBedOccupancy.get()[1] + multiBedOccupancy.get()[1] + luxuryOccupancy.get()[1];
        } catch (Exception e) {
            throw new RuntimeException("Ошибка во время многопоточного расчета занятости", e);
        }

        executor.shutdown();
        return numberOfBeds == 0 ? 0 : (float) numberOfGuests / numberOfBeds;
    }

    /**
     * Задача для расчета занятости номеров одного типа.
     * Вычисляет количество гостей и количество мест в переданных номерах.
     */
    private record RoomOccupancyTask(Map<Integer, Room> rooms) implements Callable<int[]> {

        /**
         * Метод для выполнения задачи по расчету занятости.
         * Проходит по номерам и суммирует количество гостей и количество мест.
         *
         * @return Массив из двух элементов:
         *         1) Количество гостей, зарегистрированных в номере.
         *         2) Общее количество мест в номере.
         */
        @Override
        public int[] call() {
            Iterator<Room> iterator = rooms.values().iterator();
            int numberOfGuests = 0;
            int numberOfBeds = 0;

            while (iterator.hasNext()) {
                Room room = iterator.next();
                numberOfGuests += room.getNumberOfGuests();
                numberOfBeds += room.getRoomSize();
            }

            return new int[]{numberOfGuests, numberOfBeds};
        }
    }
}
