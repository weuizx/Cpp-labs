package org.example.service.calculator;

import org.example.room.Room;

import java.util.Iterator;
import java.util.Map;

/**
 * Класс для расчета занятости номеров с использованием однопоточности.
 * Этот класс реализует интерфейс {@link OccupancyCalculator} и выполняет расчёт занятости
 * всех номеров в гостинице в одном потоке, обходя все номера и суммируя количество гостей и мест.
 */
public class SingleThreadOccupancyCalculator implements OccupancyCalculator {

    private final Map<Integer, Room> ALL_ROOMS;

    /**
     * Конструктор для создания экземпляра калькулятора с однопоточной обработкой занятости.
     *
     * @param ALL_ROOMS Карта с номерами всех типов.
     */
    public SingleThreadOccupancyCalculator(Map<Integer, Room> ALL_ROOMS) {
        this.ALL_ROOMS = ALL_ROOMS;
    }

    /**
     * Метод для расчета текущей занятости всех номеров.
     * Занятость вычисляется как отношение количества занятых мест к общему количеству мест в номерах.
     * Метод выполняется в одном потоке.
     *
     * @return Занятость в виде числа с плавающей точкой от 0.0 до 1.0.
     *         Например, если занято 50% мест, метод вернет 0.5.
     */
    @Override
    public float getOccupancy() {
        Iterator<Room> iterator = ALL_ROOMS.values().iterator();
        int numberOfGuests = 0;
        int numberOfBeds = 0;

        while (iterator.hasNext()) {
            Room room = iterator.next();
            numberOfGuests += room.getNumberOfGuests();
            numberOfBeds += room.getRoomSize();
        }

        return numberOfBeds == 0 ? 0 : (float) numberOfGuests / numberOfBeds;
    }
}
