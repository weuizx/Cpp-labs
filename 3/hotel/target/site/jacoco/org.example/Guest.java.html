package org.example;

import java.util.Date;

/**
 * Класс, представляющий гостя гостиницы.
 * Этот класс содержит информацию о госте, включая его уникальный идентификатор,
 * имя, номер комнаты, дату заселения и продолжительность пребывания.
 */
public class Guest {

    /**
     * Счётчик для генерации уникальных идентификаторов для каждого гостя.
     */
    private static int idCounter = 0;

    /**
     * Уникальный идентификатор гостя.
     */
    private final int guestId;

    /**
     * Полное имя гостя.
     */
    private final String fullName;

    /**
     * Идентификатор номера, в котором размещается гость.
     */
    private int roomId;

    /**
     * Дата заселения гостя.
     */
    private final Date checkInDate;

    /**
     * Продолжительность пребывания гостя в днях.
     */
    private final int daysOfStay;

    /**
     * Конструктор для создания нового объекта {@link Guest}.
     * Устанавливает уникальный идентификатор, имя гостя, дату заселения и продолжительность пребывания.
     *
     * @param daysOfStay Продолжительность пребывания гостя в днях.
     * @param fullName Полное имя гостя.
     */
    public Guest(int daysOfStay, String fullName) {
        this.guestId = idCounter++;
        this.fullName = fullName;
        this.checkInDate = new Date();
        this.daysOfStay = daysOfStay;
    }

    /**
     * Возвращает уникальный идентификатор гостя.
     *
     * @return Уникальный идентификатор гостя.
     */
    public int getGuestId() {
        return guestId;
    }

    /**
     * Возвращает полное имя гостя.
     *
     * @return Полное имя гостя.
     */
    public String getFullName() {
        return this.fullName;
    }

    /**
     * Возвращает идентификатор номера, в котором размещен гость.
     *
     * @return Идентификатор номера.
     */
    public int getRoomId() {
        return roomId;
    }

    /**
     * Возвращает дату заселения гостя.
     *
     * @return Дата заселения гостя.
     */
    public Date getCheckInDate() {
        return checkInDate;
    }

    /**
     * Возвращает продолжительность пребывания гостя в днях.
     *
     * @return Продолжительность пребывания в днях.
     */
    public int getDaysOfStay() {
        return daysOfStay;
    }

    /**
     * Устанавливает идентификатор номера, в котором размещен гость.
     *
     * @param roomId Идентификатор номера.
     */
    public void setRoomId(int roomId) {
        this.roomId = roomId;
    }

    /**
     * Возвращает строковое представление объекта {@link Guest}, включая информацию
     * о номере, имени, дате заселения и продолжительности пребывания.
     *
     * @return Строковое представление объекта.
     */
    @Override
    public String toString() {
        return "{" +
                "номер гостя:" + guestId +
                " ФИО гостя :" + fullName +
                ", день заселения:" + checkInDate +
                ", продолжительность аренды:" + daysOfStay +
                '}';
    }
}
