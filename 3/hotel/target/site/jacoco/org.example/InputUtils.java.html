package org.example;

import java.util.InputMismatchException;
import java.util.Scanner;

/**
 * Утилитный класс для работы с пользовательским вводом.
 * Этот класс содержит методы для чтения данных с консоли, включая полное имя пользователя и целые числа в пределах заданного диапазона.
 * Методы автоматически обрабатывают ошибки ввода и повторно запрашивают ввод в случае неверных данных.
 */
public class InputUtils {

    /**
     * Объект для считывания ввода с консоли.
     */
    private static final Scanner input = new Scanner(System.in);

    /**
     * Читает полное имя пользователя с консоли.
     * Метод ожидает, что имя будет введено в одной строке.
     *
     * @return Полное имя пользователя, введенное с консоли.
     */
    public static String readFullName() {
        String fullName;
        fullName = input.nextLine();
        return fullName;
    }

    /**
     * Читает целое число из консоли в пределах заданного диапазона.
     * Если введенное число не входит в указанный диапазон или является недопустимым, метод повторяет запрос.
     *
     * @param start Начало диапазона (включительно).
     * @param end   Конец диапазона (включительно).
     * @return Введенное число, которое находится в пределах диапазона от start до end.
     * @throws IllegalArgumentException если start больше или равен end.
     */
    public static int readIntInRange(int start, int end) {
        if (end < start) {
            throw new IllegalArgumentException("start должен быть меньше end");
        }

        int output = 0;
        boolean validInput = false;

        while (!validInput) {
            try {
                output = input.nextInt();

                if (output >= start && output <= end) {
                    validInput = true;
                } else {
                    System.out.println("Значение должно быть в диапазоне [" + start + ", " + end + "]! Попробуйте еще раз!");
                }
            } catch (InputMismatchException e) {
                System.out.println("Ошибка ввода! Попробуйте еще раз!");
                input.next();
            }
        }
        input.nextLine();

        return output;
    }
}
