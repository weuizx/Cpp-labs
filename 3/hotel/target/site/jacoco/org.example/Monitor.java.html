package org.example;

import org.example.room.LuxuryRoom;
import org.example.room.Room;
import org.example.room.RoomType;
import org.example.service.HotelService;

import java.util.*;
import java.util.stream.Collectors;

/**
 * Класс для отображения консольного интерфейса гостиничного сервиса.
 * Этот класс позволяет пользователю:
 * - Выполнять регистрацию и выселение гостей.
 * - Просматривать информацию о свободных и всех номерах.
 * - Получать подробную информацию о номере.
 * - Проверять уровень занятости гостиницы.
 */
public class Monitor {

    /**
     * Сервис гостиницы, используемый для выполнения операций.
     */
    private final HotelService hotelService = new HotelService();

    /**
     * Список доступных опций в меню.
     */
    private final List<String> options = List.of(
            "Завершить сеанс",
            "Оформить регистрацию гостя",
            "Оформить выселение гостя",
            "Вывести информацию о свободных номерах",
            "Вывести информацию о всех номерах гостиницы",
            "Вывести подробную информацию о номере",
            "Вывести информацию о занятости гостиницы");

    /**
     * Сопоставление опций с соответствующими методами для их обработки.
     */
    private final Map<Integer, Runnable> methodHolder = new HashMap<>();

    {
        methodHolder.put(1, this::processCheckIn);
        methodHolder.put(2, this::processEvict);
        methodHolder.put(3, this::processPrintFreeRooms);
        methodHolder.put(4, this::processPrintAllRooms);
        methodHolder.put(5, this::processPrintRoomInfoById);
        methodHolder.put(6, this::processGetOccupancy);
    }

    /**
     * Основной метод, управляющий циклом работы программы.
     * Отображает меню, принимает выбор пользователя и вызывает соответствующий метод.
     */
    public void process() {
        printOptions();
        int optionNumber = InputUtils.readIntInRange(0, options.size());
        if (optionNumber == 0) {
            System.out.println("Сеанс завершен. До свидания!");
            return;
        }
        methodHolder.get(optionNumber).run();
        process();
    }

    /**
     * Выводит список доступных опций для пользователя.
     */
    private void printOptions() {
        System.out.println("Введите номер опции :");
        for (int i = 0; i < options.size(); i++) {
            System.out.println(String.valueOf(i) + ". " + options.get(i));
        }
    }

    /**
     * Выводит типы доступных номеров.
     */
    private void printRoomTypes() {
        System.out.println("Введите тип номера :");
        RoomType[] roomTypes = RoomType.values();
        for (int i = 0; i < roomTypes.length; i++) {
            System.out.println(String.valueOf(i + 1) + ". " + roomTypes[i].getDescription());
        }
    }

    /**
     * Обрабатывает процесс регистрации гостя.
     * Запрашивает данные у пользователя и регистрирует гостя в соответствующем номере.
     */
    private void processCheckIn() {
        System.out.println("Введите ФО гостя :");
        String fullName = InputUtils.readFullName();
        System.out.println("Введите количество дней аренды :");
        int daysOfStay = InputUtils.readIntInRange(1, 365);
        Guest guest = new Guest(daysOfStay, fullName);

        printRoomTypes();
        int numberOfRoomType = InputUtils.readIntInRange(1, 3);
        RoomType roomType = RoomType.values()[numberOfRoomType - 1];
        try {
            guest = hotelService.checkInGuest(guest, roomType);
            System.out.println("Гость с id = " + guest.getGuestId() + " успешно зарегистрирован в номере " + guest.getRoomId());
            if (roomType.equals(RoomType.LUXURY)) {
                System.out.println("Добавить второго гостя в номер?\n1. Да\n2. Нет");
                if (InputUtils.readIntInRange(1, 2) == 1) {
                    System.out.println("Введите ФИО гостя :");
                    fullName = InputUtils.readFullName();
                    Guest guest2 = new Guest(daysOfStay, fullName);
                    LuxuryRoom room = (LuxuryRoom) hotelService.getRoomById(guest.getRoomId()).get();
                    room.checkInGuest(guest2);
                    System.out.println("Гость с id = " + guest2.getGuestId() + " успешно зарегистрирован в номере " + guest2.getRoomId());
                }
            }
        } catch (Exception e) {
            System.out.println("Не удалось заселить гостя. " + e.getMessage());
        }
    }

    /**
     * Обрабатывает процесс выселения гостя.
     * Запрашивает id гостя, выселяет его из номера и выводит сумму, которую он должен заплатить.
     */
    private void processEvict() {
        System.out.println("Введите id гостя");
        int guestId = InputUtils.readIntInRange(0, Integer.MAX_VALUE);
        Optional<Guest> optionalGuest = hotelService.getGuestById(guestId);
        int price;
        if (optionalGuest.isPresent()) {
            price = hotelService.evictGuest(optionalGuest.get());
            System.out.println("Гость успешно выселен, к оплате " + price);
        } else {
            System.out.println("Гость с id " + guestId + " не найден");
        }
    }

    /**
     * Обрабатывает процесс вывода информации о свободных номерах.
     */
    private void processPrintFreeRooms() {
        List<Room> rooms = hotelService.getFreeRooms();
        System.out.println("Свободные номера :");
        System.out.println(rooms.stream().
                map(room -> ("{Номер:" + room.getRoomId() + ", тип:" + room.getRoomType().getDescription() + "}")).
                collect(Collectors.joining("\n")));
    }

    /**
     * Обрабатывает процесс вывода информации о всех номерах гостиницы.
     */
    private void processPrintAllRooms() {
        List<Room> rooms = hotelService.getAllRooms();
        System.out.println("Номера гостиницы :");
        System.out.println(rooms.stream().
                map(room -> ("{Номер:" + room.getRoomId() + ", тип:" + room.getRoomType().getDescription() + "}")).
                collect(Collectors.joining("\n")));
    }

    /**
     * Обрабатывает процесс вывода подробной информации о номере по его id.
     */
    private void processPrintRoomInfoById() {
        System.out.println("Введите id номера");
        int roomId = InputUtils.readIntInRange(0, Integer.MAX_VALUE);
        Optional<Room> optionalRoom = hotelService.getRoomById(roomId);
        if (optionalRoom.isPresent()) {
            System.out.println(optionalRoom.get());
        } else {
            System.out.println("Номер с id " + roomId + " не найден");
        }
    }

    /**
     * Обрабатывает процесс вывода информации о занятости гостиницы.
     */
    private void processGetOccupancy() {
        System.out.println("Занятость гостиницы : " + Math.round(hotelService.getOccupancy() * 100) + "%");
    }

    /**
     * Основной метод программы.
     * Запускает процесс работы с гостиницей.
     */
    public static void main(String[] args) {
        Monitor monitor = new Monitor();
        monitor.process();
    }
}
